初始化投票系统

本系统使用 Supabase 免费数据库（非 Google），
配置完成后分享链接即可多人参与投票。

1
创建 Supabase 项目
打开 supabase.com → 注册账号（支持邮箱注册）→ 点击「New project」→ 填写项目名和数据库密码 → 等待约 1 分钟创建完成。
2
运行初始化 SQL
在 Supabase 左侧菜单点击「SQL Editor」→ 点击「New query」→ 粘贴以下 SQL → 点击「Run」执行：
-- 创建选项表 CREATE TABLE IF NOT EXISTS vote_options ( id TEXT PRIMARY KEY DEFAULT gen_random_uuid()::TEXT, name TEXT NOT NULL, link TEXT DEFAULT '', image TEXT DEFAULT '', votes INTEGER DEFAULT 0, sort_order BIGINT DEFAULT 0 ); -- 创建设置表 CREATE TABLE IF NOT EXISTS vote_meta ( key TEXT PRIMARY KEY, value TEXT ); -- 写入默认设置 INSERT INTO vote_meta (key, value) VALUES ('title', '🗳️ 在线投票'), ('desc', '每天3票，每票投不同选项'), ('dailyLimit', '3') ON CONFLICT (key) DO NOTHING; -- 原子计票函数（防并发冲突） CREATE OR REPLACE FUNCTION increment_vote(option_id TEXT) RETURNS void LANGUAGE SQL AS $$ UPDATE vote_options SET votes = votes + 1 WHERE id = option_id; $$; -- 开放读写权限 ALTER TABLE vote_options ENABLE ROW LEVEL SECURITY; ALTER TABLE vote_meta ENABLE ROW LEVEL SECURITY; DROP POLICY IF EXISTS "pub_read_opt" ON vote_options; DROP POLICY IF EXISTS "pub_write_opt" ON vote_options; DROP POLICY IF EXISTS "pub_read_meta" ON vote_meta; DROP POLICY IF EXISTS "pub_write_meta" ON vote_meta; CREATE POLICY "pub_read_opt" ON vote_options FOR SELECT USING (true); CREATE POLICY "pub_write_opt" ON vote_options FOR ALL USING (true); CREATE POLICY "pub_read_meta" ON vote_meta FOR SELECT USING (true); CREATE POLICY "pub_write_meta" ON vote_meta FOR ALL USING (true);
